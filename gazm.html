Adatbázisok létrehozása

Táblák létrehozása:

  • CREATE TABLE 212e.paciensek ( pid VARCHAR(11) PRIMARY KEY, nev VARCHAR(100), cim VARCHAR(100), telefon VARCHAR(20) );
  • CREATE TABLE 212e.vizsgalat ( vid INT, datum DATE, pid VARCHAR(11), PRIMARY KEY (vid, datum), FOREIGN KEY (pid) REFERENCES paciensek(pid) );
  • CREATE TABLE 212e.betegseg ( bnev VARCHAR(50) PRIMARY KEY );
  • CREATE TABLE 212e.gyogyszer ( gynev VARCHAR(50) PRIMARY KEY );
  • CREATE TABLE 212e.diagnózis ( vid INT, datum DATE, bnev VARCHAR(50), PRIMARY KEY (vid, datum, bnev), FOREIGN KEY (vid, datum) REFERENCES vizsgalat(vid, datum), FOREIGN KEY (bnev) REFERENCES betegseg(bnev) );
